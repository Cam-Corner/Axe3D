#version 330 core

layout (location = 0) in vec3 LPos;
layout (location = 1) in vec3 LNormal;
layout (location = 2) in vec2 LTextCoords;

uniform mat4 Model;
uniform mat4 View;
uniform mat4 Projection;

out vec2 TextCoords;
out vec3 VertexPositionPassed;
out vec3 VertexNormalPassed;

void main()
{
	
	VertexNormalPassed = mat3(Model) * normalize(LNormal);
	VertexPositionPassed = mat4x3(Model) * vec4(LPos, 1.0f);
	TextCoords = LTextCoords;
	
	gl_Position = Projection * View * Model * vec4(LPos, 1.0f);
}
